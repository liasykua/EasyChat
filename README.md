```yaml
# =====================================================
#                   Налаштування EasyChat
# =====================================================
#
# Доступні плейсхолдери у рядках формату:
#   {prefix}    - префікс гравця з LuckPerms
#   {suffix}    - суфікс гравця з LuckPerms
#   {player}    - нікнейм гравця
#   {message}   - повідомлення в чаті
#   {sender}    - нікнейм відправника (приватний чат)
#   {recipient} - нікнейм отримувача (приватний чат)
#   {chat}      - повністю відформатоване повідомлення (social-spy)
#
# Кольорові коди: використовуйте & за яким йде код кольору (наприклад &a, &6, &8)
# Hex кольори: &#RRGGBB (наприклад &#ff5500)
#
# =====================================================
#                    Права доступу
# =====================================================
#
#   easychat.chat.local    - писати в локальному чаті
#   easychat.chat.global   - писати в глобальному чаті (префікс !)
#   easychat.chat.personal - використовувати /msg та /reply
#   easychat.spy           - вмикати соціальне шпигунство (/spy)
#   easychat.clearchat     - очищати чат (/clearchat)
#   easychat.reload        - перезавантажувати плагін (/easychatreload)
#
# =====================================================

chats:

  # Глобальний чат — видно всім гравцям на сервері.
  # Гравці активують його, починаючи повідомлення з визначеного символу.
  # Встановіть radius: -1 для необмеженої дальності.
  global:
    format: "&6[G] {prefix}{player}{suffix} &8» &f{message}"
    symbol: "!"
    radius: "-1"

  # Локальний чат — видно лише гравцям у межах заданої відстані (у блоках).
  # Це чат за замовчуванням, якщо не використовується символ.
  local:
    format: "&a[L] {prefix}{player}{suffix} &8» &f{message}"
    radius: "100"

  # Приватний чат — особисті повідомлення між двома гравцями через /msg.
  personal:
    format: "&b[P] &f{sender} &7-> &f{recipient} &8» &f{message}"

  # Соціальне шпигунство — формат для показу перехоплених приватних повідомлень
  # гравцям, у яких увімкнено режим шпигунства (/spy).
  social-spy:
    format: "&e[S] {chat}"

# =====================================================
#                    Повідомлення
# =====================================================

messages:
  no-permission: "&cУ вас немає прав для цієї дії."
  player-not-found: "&cГравця не знайдено."
  no-reply-target: "&cВам немає кому відповісти."
  msg-usage: "&cВикористання: /msg <гравець> <повідомлення>"
  reply-usage: "&cВикористання: /reply <повідомлення>"
  ignore-usage: "&cВикористання: /ignore <гравець>"
  ignore-added: "&7Ви тепер ігноруєте &c{player}&7."
  ignore-removed: "&7Ви більше не ігноруєте &a{player}&7."
  ignored-by-player: "&cЦей гравець ігнорує вас."
  you-are-ignored: "&cВи ігноруєте цього гравця."
  spy-enabled: "&eСоціальне шпигунство &aувімкнено&e."
  spy-disabled: "&eСоціальне шпигунство &cвимкнено&e."
  chat-cleared: "&aЧат було очищено гравцем &e{player}&a."
  plugin-reloaded: "&aНалаштування EasyChat перезавантажено."
  no-local-permission: "&cУ вас немає прав писати в локальному чаті."
  no-global-permission: "&cУ вас немає прав писати в глобальному чаті."
  cannot-msg-self: "&cВи не можете надіслати повідомлення самому собі."
```
